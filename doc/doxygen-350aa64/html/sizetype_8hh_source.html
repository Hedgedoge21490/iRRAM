102 // sizetype_normalize(e) **********************************************
103 // Try to keep the mantissa between max_mantissa and min_mantissa.
104 // It may not be larger than max_mantissa.
105 // It still may stay smaller than min_mantissa afterwards.
106 // The value of e may slighty increase during this operation!
107 // The value of e must never decrease!
108 // If the resulting exponent is out of range (MP_min,MP_max)
109 // correct it with an increase in value or produce an error.
110 // ********************************************************************
111 
112 
113 inline void sizetype_normalize( sizetype& e) {
114  if (iRRAM_unlikely(e.mantissa < min_mantissa)) {
115  e.mantissa <<= BIT_RANGE;
116  e.exponent -= BIT_RANGE;
117  }
118  if (iRRAM_unlikely( e.mantissa >= max_mantissa ) ) {
119  e.mantissa = ( e.mantissa>> DIFF_BITS ) + 1;
120  e.exponent += DIFF_BITS;
121  }
122  if (iRRAM_unlikely( e.exponent < MP_min ) ){
123  e.exponent = min_exponent;
124  }
125  if (iRRAM_unlikely( e.exponent >= MP_max ) )
126  {
127  iRRAM_DEBUG1(1,"exponent too big in sizetype_normalize ");
128  REITERATE(0);
129  }
130 }

150 inline void sizetype_inc(sizetype& x,const sizetype& y)
151 {
152  if ( y.exponent > x.exponent) {
153  x.mantissa=scale(x.mantissa,(y.exponent-x.exponent))
154  +y.mantissa +1 ;
155  x.exponent=y.exponent;

183 inline void sizetype_add_one(sizetype& x,const sizetype& y,const SIZETYPEEXPONENT zexp)
184 {
185 // replace 1*2^(-zexp) by 1048576*2^(-zexp-20)
186 // this is still the same numerical value!
187 // if zexp is larger than y.exponent, the result is closer to the intended result
188  SIZETYPEEXPONENT zexp_scale= zexp-20;
189  if ( y.exponent > zexp_scale) {
190  x.exponent= y.exponent;
191  x.mantissa= y.mantissa + 1048576 ;
192  } else {
193  x.exponent= zexp_scale;
194  x.mantissa= (scale(y.mantissa,(zexp_scale-y.exponent))+1) + 1048576 ;
195  }
196  sizetype_normalize(x);
197 }
