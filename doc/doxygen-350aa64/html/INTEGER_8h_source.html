184 inline INTEGER scale(const INTEGER& x, const int n)
185 {
186  if (n==0) return x;
187  if (n==1) return 2*x;
188  INTEGER y=1;
189  INTEGER xc=2;
190  if (n<0) {xc=y/2;n=-n;}
191  if (n==1) return xc*x;
192  for (int k=n;k>0;k=k/2)
193  {
194  if (k%2==1) y*=xc;
195  if (k==1) break;
196  xc*=xc;
197  }
198  return y*x;
199 }

282 inline INTEGER operator ^ (const INTEGER& x, const INTEGER& y)
283 {
284  if (y==0 || x==1) return 1;
285  if (y==1) return x;

288  return x^(unsigned int)(MP_INTEGER_to_int(y.value));
