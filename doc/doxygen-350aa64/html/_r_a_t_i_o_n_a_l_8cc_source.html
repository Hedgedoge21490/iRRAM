159 RATIONAL scale(RATIONAL x, int n)
160 {
161  if (n==0) return x;
162  if (n==1) { x *= 2; return x; }
163  RATIONAL y=1;
164  RATIONAL xc=2;
165  if (n<0) {xc=y/2;n=-n;}
166  if (n==1) { xc *= x; return xc; }
167  for (int k=n;k>0;k=k/2)
168  {
169  if (k%2==1) y*=xc;
170  if (k==1) break;
171  xc*=xc;
172  }
173  y *= x;
174  return y;
