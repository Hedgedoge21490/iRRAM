473 inline REAL & REAL::operator=(const REAL & y)
474 {
475  if (iRRAM_unlikely(value || y.value)) {
476  if (value && y.value) {
477  this->mp_copy(y);
478  return (*this);
479  }
480  if (y.value) {
481  if (state.ACTUAL_STACK.prec_step == 0)
482  this->mp_from_mp(y);
483  else
484  this->mp_copy_init(y);
485  return *this;
486  }
487  dp = y.dp;
488  mp_make_mp();
489  return *this;
490  }
491  dp = y.dp;
492  return *this;
493 }

619 inline REAL operator*(const REAL & x, const REAL & y)
620 {
621  if (iRRAM_unlikely(x.value || y.value))
622  return x.mp_conv().mp_multiplication(y.mp_conv());
623  REAL::double_pair z;
624  if (x.dp.lower_pos >= 0 && y.dp.lower_pos >= 0) {
625  z.lower_pos = x.dp.lower_pos * y.dp.lower_pos;
626  z.upper_neg = (-x.dp.upper_neg) * y.dp.upper_neg;
627  } else if (x.dp.upper_neg >= 0 && y.dp.upper_neg >= 0) {
628  z.lower_pos = x.dp.upper_neg * y.dp.upper_neg;
629  z.upper_neg = (-x.dp.lower_pos) * y.dp.lower_pos;
630  } else if (x.dp.upper_neg >= 0 && y.dp.lower_pos >= 0) {
631  z.lower_pos = x.dp.lower_pos * (-y.dp.upper_neg);
632  z.upper_neg = x.dp.upper_neg * y.dp.lower_pos;
633  } else if (x.dp.lower_pos >= 0 && y.dp.upper_neg >= 0) {
634  z.lower_pos = (-x.dp.upper_neg) * y.dp.lower_pos;
635  z.upper_neg = x.dp.lower_pos * y.dp.upper_neg;
636  } else {
637  z.lower_pos = fmin((-x.dp.upper_neg) * y.dp.lower_pos,
638  x.dp.lower_pos * (-y.dp.upper_neg));
639  z.upper_neg = fmin((-x.dp.upper_neg) * y.dp.upper_neg,
640  x.dp.lower_pos * (-y.dp.lower_pos));
641  }
642  return REAL(z);
643 }

661 inline REAL & REAL::operator*=(int n)
662 {
663  if (iRRAM_unlikely(value))
664  return *this = mp_multiplication(n);
665  if (n >= 0) {
666  dp.lower_pos *= n;
667  dp.upper_neg *= n;
668  } else {
669  double tmp = -dp.upper_neg * n;

671  dp.lower_pos = tmp;
672  }
