71 inline void sizetype_normalize( sizetype& e) {
72  if (iRRAM_unlikely(e.mantissa < min_mantissa)) {
73  e.mantissa <<= BIT_RANGE;
74  e.exponent -= BIT_RANGE;
75  }
76  if (iRRAM_unlikely( e.mantissa >= max_mantissa ) ) {
77  e.mantissa = ( e.mantissa>> DIFF_BITS ) + 1;
78  e.exponent += DIFF_BITS;
79  }
80  if (iRRAM_unlikely( e.exponent < MP_min ) ){
81  e.exponent = min_exponent;
82  }
83  if (iRRAM_unlikely( e.exponent >= MP_max ) )
84  {
85  iRRAM_DEBUG1(1,"exponent too big in sizetype_normalize ");
86  REITERATE(0);
87  }
88 }

120 {
121  if ( y.exponent > x.exponent) {
122  x.mantissa=scale(x.mantissa,(y.exponent-x.exponent))
123  +y.mantissa +1 ;

158 inline void sizetype_add_one(sizetype& x, const sizetype& y, typename sizetype::exponent_t zexp)
159 {
160 // replace 1*2^(-zexp) by 1048576*2^(-zexp-20)
161 // this is still the same numerical value!
162 // if zexp is larger than y.exponent, the result is closer to the intended result
163  typename sizetype::exponent_t zexp_scale= zexp-20;
164  if ( y.exponent > zexp_scale) {
165  x.exponent= y.exponent;
166  x.mantissa= y.mantissa + 1048576 ;
167  } else {
168  x.exponent= zexp_scale;
169  x.mantissa= (scale(y.mantissa,(zexp_scale-y.exponent))+1) + 1048576 ;
170  }
171  sizetype_normalize(x);
172 }

347 { unsigned long long lmantissa=
348  ((unsigned long long)(y.mantissa))*z.mantissa;
349  x.exponent=y.exponent+z.exponent;

439 inline bool sizetype_less(const sizetype& y,const sizetype& z)
440 { unsigned int mantissa;
441  if (iRRAM_unlikely(y.mantissa==0)) return true;
442  if (iRRAM_unlikely(z.mantissa==0)) return false;
443  if (y.exponent>z.exponent)
444  {
445  mantissa=scale(z.mantissa,y.exponent-z.exponent);
446  return (mantissa>=y.mantissa);
447  } else {
448  mantissa=scale(y.mantissa,z.exponent-y.exponent);
449  return (mantissa<z.mantissa);
450  }
451 }
