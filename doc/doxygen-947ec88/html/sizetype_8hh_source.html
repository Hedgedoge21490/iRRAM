72 inline void sizetype_normalize( sizetype& e) {
73  if (iRRAM_unlikely(e.mantissa < min_mantissa)) {
74  e.mantissa <<= BIT_RANGE;
75  e.exponent -= BIT_RANGE;
76  }
77  if (iRRAM_unlikely( e.mantissa >= max_mantissa ) ) {
78  e.mantissa = ( e.mantissa>> DIFF_BITS ) + 1;
79  e.exponent += DIFF_BITS;
80  }
81  if (iRRAM_unlikely( e.exponent < MP_min ) ){
82  e.exponent = min_exponent;
83  }
84  if (iRRAM_unlikely( e.exponent >= MP_max ) )
85  {
86  iRRAM_DEBUG1(1,"exponent too big in sizetype_normalize ");
87  REITERATE(0);
88  }
89 }

159 inline void sizetype_add_one(sizetype& x, const sizetype& y, typename sizetype::exponent_t zexp)
160 {
161 // replace 1*2^(-zexp) by 1048576*2^(-zexp-20)
162 // this is still the same numerical value!
163 // if zexp is larger than y.exponent, the result is closer to the intended result
164  typename sizetype::exponent_t zexp_scale= zexp-20;
165  if ( y.exponent > zexp_scale) {
166  x.exponent= y.exponent;
167  x.mantissa= y.mantissa + 1048576 ;
168  } else {
169  x.exponent= zexp_scale;
170  x.mantissa= (scale(y.mantissa,(zexp_scale-y.exponent))+1) + 1048576 ;
171  }
172  sizetype_normalize(x);
173 }

442 inline bool sizetype_less(const sizetype& y,const sizetype& z)
443 { unsigned int mantissa;
444  if (iRRAM_unlikely(y.mantissa==0)) return true;
445  if (iRRAM_unlikely(z.mantissa==0)) return false;
446  if (y.exponent>z.exponent)
447  {
448  mantissa=scale(z.mantissa,y.exponent-z.exponent);
449  return (mantissa>=y.mantissa);
450  } else {
451  mantissa=scale(y.mantissa,z.exponent-y.exponent);
452  return (mantissa<z.mantissa);
453  }

469 inline int sizetype_log2(const sizetype &x)
470 {
471  int r;
472  typename sizetype::mantissa_t v = x.mantissa;
473  if (!v)
474  return min_exponent;
475  r = CHAR_BIT * sizeof(v) - clz(v) - 1;
476  r += (v & (v-1)) != 0; /* ceil(log2(v)) */
477  r += x.exponent;
