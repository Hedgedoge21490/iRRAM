290 inline ext_mpfr_type ext_mpfr_init()
291 {
292  ext_mpfr_type z;
293  if (mpfr_FreeVarCount > 0) {
294  mpfr_FreeVarCount -= 1;
295  z = mpfr_FreeVars[mpfr_FreeVarCount];
296  } else {
297  z = (ext_mpfr_type)malloc(sizeof(__mpfr_struct));
298  mpfr_init(z);
299  mpfr_TotalAllocVarCount++;
300  }
301  ext_mpfr_var_count += 1;
302 
303  /* fprintf(stderr,"create %x\n",z); */
304 
305  return z;
306 }

310 inline void ext_mpfr_free(ext_mpfr_type z)
311 {
312  /* fprintf(stderr,"delete %x\n",z); */
313  if (mpfr_FreeVarCount < MaxFreeVars) {
314  mpfr_FreeVars[mpfr_FreeVarCount] = z;
315  mpfr_FreeVarCount += 1;
316  } else {
317  mpfr_clear(z);
318  free(z);
319  mpfr_TotalFreedVarCount++;
320  }
321  ext_mpfr_var_count -= 1;
322 }
