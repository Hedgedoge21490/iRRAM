94 iRRAM_STATIC inline void ext_mpfr_remove_trailing_zeroes (mpfr_t x)
95 {
96  unsigned int xn = MPFR_MSW_INDEX(x);
97  unsigned int i;
98  unsigned int j=(xn+1)*BITS_PER_MP_LIMB;
99  mp_limb_t *xp = MPFR_MANT(x);
100 
101  if ( xp[0] != 0 ) return;
102  /* first determine last non-zero limb */
103  for (i=0; i<=xn && xp[i]==0; (i++) ) j-=BITS_PER_MP_LIMB;
104 
105  if (i > xn) return;
106  mpfr_round_prec (x, GMP_RNDN, j);
107 }

130 iRRAM_STATIC inline void ext_mpfr_getsize(ext_mpfr_type z,ext_mpfr_sizetype* s)
131 {
132  int zn=MPFR_MSW_INDEX(z);
133  if ( MPFR_NOTZERO(z) ) {
134 #if BITS_PER_MP_LIMB == 32
135  s->mantissa=( ((z->_mpfr_d)[zn]) >> 1 ) + 1;
136  s->exponent=MPFR_EXP(z)-BITS_PER_MP_LIMB+1;
137 #elif BITS_PER_MP_LIMB == 64
138  s->mantissa=( ((z->_mpfr_d)[zn]) >> (BITS_PER_MP_LIMB/2+1) ) + 1;
139  s->exponent=MPFR_EXP(z)- (BITS_PER_MP_LIMB/2) +1;
140 #else
141 # error unable to handle BITS_PER_MP_LIMP != 32 and != 64
142 #endif
143  } else {
144  s->mantissa=0;
145  s->exponent=GMP_min;
146  }
147 }

307 inline char* ext_mpfr_swritee(ext_mpfr_type z,int w)
308 {
309  char* s=(char*)(malloc(w+1));
310  char* r=&s[2];
311  long e, e_abs,r_len;
312  int i;
313  int flag=0;
314  int sgn = mpfr_sgn (z);
315 
316  mpfr_t z_copy;
317  mpfr_init2(z_copy,mpfr_get_prec(z));
318  mpfr_abs(z_copy,z,iRRAM_mpfr_rounding_mode);
319  mpfr_get_str (r, &e, 10, w-6,z_copy,iRRAM_mpfr_rounding_mode);
320  mpfr_clear(z_copy);
321 
322  /* in case trailing zeroes were discarded, we add them again... */
323  for (i=0;i<w-6;i++) {
324  if ( r[i]==0 ) flag=1 ;
325  if ( flag ) r[i]='0';
326  }
327 
328  /*
329  we want to have an error of less than 0.51ulp, so above
330  we took two additional digits giving an error of at most 0.01ulp
331  (concerning the intended precision).
332  The position of these digits is (w-8,w-7) for the
333  usual 4-decimal exponents and decreases for the larger cases.
334  If the first of the two additional digits is 5 or greater,
335  we round upwards (giving at most additional 0.5ulp)
336  */
337 
338  r_len=e; if (r_len<0) r_len=-r_len;
339  if (r_len<10000){r_len=4;}
340  else if (r_len<100000){r_len=5;}
341  else if (r_len<1000000){r_len=6;}
342  else if (r_len<10000000){r_len=7;}
343  else if (r_len<100000000){r_len=8;}
344  else if (r_len<1000000000){r_len=9;}
345  else {r_len=10;};
346 
347  /*
348  r_len is the number of decimal places of the exponent prior to the rounding
349  */
350 
351 
352  if ( w < r_len+5) w=r_len+5;
353  /*
354  Produce at least one significant digit...
355  */
356 
357  if ( r[w-r_len-4]>='5' ){
358  for (i=w-r_len-5;i>=0;i--){
359  if ( r[i] < '9' ) {
360  r[i]++;
361  break;
362  }
363  r[i]='0';
364  }
365  /*
366  In the rare case that we switch from 0.999...999 (length n) to 1.000...000 (length n+1),
367  we have to correct the exponent and the mantissa, but changing the mantissa is simple:
368  (1) If the (decimal) length of the exponent does not change, result is 0.100...000 (length n)
369  (2) If the (decimal) length of the exponent increases, result is 0.100...000 (length n-1)
370  (3) If the (decimal) length of the exponent decreases, result is 0.100...000 (length n+1)
371  So we simply set the last decimal to '0'. The 0.51ulp accuracy is true in all three cases!
372  */
373  if (i<0){
374  r[0]='1';
375  r[w-r_len-4]='0';
376  e++;
377  }
378  }
379  r[w-r_len-3] = 0;
380  if (sgn== -1) s[0]='-';
381  else s[0]='+';
382  s[1]='.';
383 
384  e_abs=e; if (e_abs<0) e_abs=-e_abs;
385  if (e_abs<10000){e_abs=4;}
386  else if (e_abs<100000){e_abs=5;}
387  else if (e_abs<1000000){e_abs=6;}
388  else if (e_abs<10000000){e_abs=7;}
389  else if (e_abs<100000000){e_abs=8;}
390  else if (e_abs<1000000000){e_abs=9;}
391  else {e_abs=10;};
392  if (e_abs > w-2){sprintf(&s[w-6],"E*****");
393  } else {
394  if (e_abs==4){sprintf(&s[w-6],"E%+05ld",e);}
395  else if (e_abs==5){sprintf(&s[w-7],"E%+06ld",e);}
396  else if (e_abs==6){sprintf(&s[w-8],"E%+07ld",e);}
397  else if (e_abs==7){sprintf(&s[w-9],"E%+08ld",e);}
398  else if (e_abs==8){sprintf(&s[w-10],"E%+09ld",e);}
399  else if (e_abs==9){sprintf(&s[w-11],"E%+010ld",e);}
400  else {sprintf(&s[w-12],"E%+011ld",e);};
401  }
402  return s;
403 }
