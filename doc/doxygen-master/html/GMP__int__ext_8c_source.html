133 /**************************************************************************/
134 /* writes integer to a string */
135 /* writes z to string with length w */
136 /* Arguments: 1. GMP_integer */
137 /* 2. int integer */
138 /* return: char* */
139 /**************************************************************************/
140 
141 
142 char* int_gmp_swritee(int_gmp_type z, int w)
143 {
144  char *n, *s;
145  int l;
146  if (w<1)w=1;
147  s=malloc(w+1);
148  n=mpz_get_str(NULL, 10, z);
149  l=strlen(n);
150  if (l>w){
151  strncpy(s,n,w);
152  s[w-1]='*';
153  s[w]='\0';
154  }else{
155  memset(s,' ',w-l);
156  strncpy(&s[w-l],n,l);
157  s[w]='\0';
158  }
159  free(n);
160  return s;
161 }
